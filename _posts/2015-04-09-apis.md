~~~antlr

/*
 * math.nez
 * basic mathematical operator
 * author: Kimio Kuramitsu
 */

File
	= Expr .*

Expr
	= Sum

Sum
	= Product {@ ( '+' #Add / '-' #Sub ) @Product }*

Product
	= Value {@ ( '*' #Mul / '/' #Div / '%' #Mod ) @Value }*

Value
	= { [0-9]+ #Int }
	/ '(' Expr ')'

~~~
